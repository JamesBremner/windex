542  double tick;

559  vl.push_back( v );
560  tick += inc;
561  if( tick >= mx )
562  break;
563  }
564  vl.push_back( mx );
565  return vl;
566  }

578  double order = ::pow(10., n - d);
579  std::stringstream ss;
