522  virtual bool WindowMessageHandler( HWND hwnd, UINT uMsg, WPARAM wParam, LPARAM lParam)
523  {
524  //std::cout << " widget " << myText << " WindowMessageHandler " << uMsg << "\n";
525  if( hwnd == myHandle )
526  {
527  switch (uMsg)
528  {
529  case WM_DESTROY:
530  // if this is the appliction window
531  // then quit application when destroyed
532  if( myID == 1 )
533  PostQuitMessage(0);
534  return true;
535  return true;
536 
537  case WM_NCDESTROY:
538  // all the children are gone
539  // so a modal display can close
540  myfModal = false;
541  return false;
542 
543  case WM_ERASEBKGND:
544  {
545 // if( ! myParent )

548  GetWindowRect(hwnd, &rc);

554 // GetClientRect(hwnd,&rc );
555 // //FillRect((HDC)wParam, &rc, myBGBrush );
556  return true;
557  }
558 
559  case WM_PAINT:
560  {
561  PAINTSTRUCT ps;
562  BeginPaint(myHandle, &ps);
563  FillRect(ps.hdc, &ps.rcPaint, myBGBrush );
564  draw(ps);
565 
566  EndPaint(myHandle, &ps);
567  }
568  return true;
569 
570  case WM_LBUTTONDOWN:
571  if( myEvents.onLeftdown() )
572  return true;
573  // the event was not completely handled, maybe the parent can look after it
574  if( myParent )
575  {
576  if( myParent->WindowMessageHandler(
577  myParent->handle(),
578  uMsg, wParam, lParam ))
579  return true;
580  }
581  break;
582 
583  case WM_MOUSEMOVE:
584  myEvents.onMouseMove( wParam, lParam );
585  break;
586 
587  case WM_MOUSEWHEEL:
588  {
589  int d = HIWORD(wParam);
590  if( d > 0xEFFF)
591  d = -120;
592  myEvents.onMouseWheel( d );
593  }
594  break;
595 
596  case WM_SIZE:
597  if( wParam == SIZE_RESTORED)
598  {
599  myEvents.onResize( LOWORD(lParam), HIWORD(lParam) );
600  return true;
601  }
602  return false;
603 
604  case WM_HSCROLL:
605  myEvents.onScrollH( LOWORD (wParam) );
606  return true;
607  case WM_VSCROLL:
608  myEvents.onScrollV( LOWORD (wParam) );
609  return true;
610 
611  case WM_COMMAND:
612  events().onMenuCommand( wParam );
613  return true;
614  }
615 
616  }
617  else
618  {
619  for( auto w : myChild )
620  {
621  if( w->WindowMessageHandler( hwnd, uMsg, wParam, lParam ))
622  return true;
623  }
624  }
