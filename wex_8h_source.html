657  virtual bool WindowMessageHandler( HWND hwnd, UINT uMsg, WPARAM wParam, LPARAM lParam)
658  {
659  //std::cout << " widget " << myText << " WindowMessageHandler " << uMsg << "\n";
660  if( hwnd == myHandle )
661  {
662  switch (uMsg)
663  {
664  case WM_DESTROY:
665  // if this is the appliction window
666  // then quit application when destroyed
667  if( myID == 1 )
668  PostQuitMessage(0);
669  return true;
670  return true;
671 
672  case WM_NCDESTROY:
673  // all the children are gone
674  // so a modal display can close
675  myfModal = false;
676  return false;
677 
678  case WM_ERASEBKGND:
679  {
680 // if( ! myParent )
681 // {
682  RECT rc;
683  GetWindowRect(hwnd, &rc);
684  FillRect((HDC)wParam, &rc, myBGBrush );
685  return true;
686 // }
687 // RECT rc;
688 // GetWindowRect(hwnd, &rc);
689 // GetClientRect(hwnd,&rc );
690 // //FillRect((HDC)wParam, &rc, myBGBrush );
691  return true;
692  }
693 
694  case WM_PAINT:
695  {
696  PAINTSTRUCT ps;
697  BeginPaint(myHandle, &ps);
698  FillRect(ps.hdc, &ps.rcPaint, myBGBrush );
699  draw(ps);
700 
701  EndPaint(myHandle, &ps);
702  }
703  return true;
704 
705  case WM_LBUTTONDOWN:
706  case WM_RBUTTONDOWN:
707  //std::cout << "click on " << myText << "\n";
708  if( myEvents.onLeftdown() )
709  return true;
710  // the event was not completely handled, maybe the parent can look after it
711  if( myParent )
712  {
713  if( myParent->WindowMessageHandler(
714  myParent->handle(),
715  uMsg, wParam, lParam ))
716  return true;
717  }
718  break;
719 
720  case WM_LBUTTONUP:
721  case WM_RBUTTONUP:
722  myEvents.onMouseUp();
723  return true;
724 
725  case WM_MOUSEMOVE:
726  myEvents.onMouseMove( wParam, lParam );
727  break;
728 
729  case WM_MOUSEWHEEL:
730  {
731  int d = HIWORD(wParam);
732  if( d > 0xEFFF)
733  d = -120;
734  myEvents.onMouseWheel( d );
735  }
736  break;
737 
738  case WM_SIZE:
739  if( wParam == SIZE_RESTORED)
740  {
741  myEvents.onResize( LOWORD(lParam), HIWORD(lParam) );
742  return true;
743  }
744  return false;
745 
746  case WM_HSCROLL:
747  if( lParam )
748  {
749  if( LOWORD(wParam) == SB_THUMBPOSITION )
750  {
751  // this gui element has received a notification
752  // that a slider has been dragged and released.
753  // The slider is a child of this gui element
754  // so find which child and call its slid event handler
755  // with the new position
756  for( auto c : myChild )
757  {
758  if( c->handle() == (HWND)lParam )
759  {
760  c->events().onSlid( HIWORD(wParam) );
761  return true;
762  }
763  }
764  }
765  return false;
766  }
767  myEvents.onScrollH( LOWORD (wParam) );
768  return true;
769  case WM_VSCROLL:
770  if( lParam )
771  {
772  if( LOWORD(wParam) == SB_THUMBPOSITION )
773  {
774  for( auto c : myChild )
775  {
776  if( c->handle() == (HWND)lParam )
777  {
778  c->events().onSlid( HIWORD(wParam) );
779  return true;
780  }
781  }
782  }
783  return false;
784  }
785  myEvents.onScrollV( LOWORD (wParam) );
786  return true;
787 
788  case WM_COMMAND:
789  if( lParam )
790  {
791  if( HIWORD(wParam) == CBN_SELCHANGE )
792  {
793  return events().onSelect( LOWORD(wParam) );
794  }
795  return true;
796  }
797  events().onMenuCommand( wParam );
798  return true;
799 
800  case WM_TIMER:
801  events().onTimer();
802  return true;
803  }
804 
805  }
806  else
807  {
808  for( auto w : myChild )
809  {
810  if( w->WindowMessageHandler( hwnd, uMsg, wParam, lParam ))
811  return true;
812  }
813  }
814 
815  return false;
816  }
