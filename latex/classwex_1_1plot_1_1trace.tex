\doxysubsection{Detailed Description}
Single trace to be plotted. 

Application code should not attempt to construct a trace Rather call one of plot\+::\+Add\+Point\+Trace, \mbox{\hyperlink{classwex_1_1plot_1_1plot_a9ff35dea82800622757536cb4e2df627}{plot\+::\+Add\+Real\+Time\+Trace}} or \mbox{\hyperlink{classwex_1_1plot_1_1plot_af55c5abc73bee295c1cdba6119b99b8c}{plot\+::\+Add\+Static\+Trace}} which return a reference to the trace which can be configured and be populated with data.


\begin{DoxyPre}
    form fm;

     construct plot to be drawn on form

    \mbox{\hyperlink{classwex_1_1plot_1_1plot}{plot::plot}} thePlot( fm );

     construct plot trace

    auto t1 = thePlot.AddStaticTrace();

     provide some data for  trace

    std::vector< double > d1 \{ 10, 15, 20, 25, 30, 25, 20, 15, 10 \};
    t1.set( d1 );

     plot in blue

    t1.color( colors::blue );

     show and run

    fm.show();
    exec();
\end{DoxyPre}
