    // construct top level window
    gui\& form = \mbox{\hyperlink{classwex_1_1maker_aee4687ec851077556856d959d74c40d4}{wex::maker::make()}};
    form.move(\{ 50,50,400,400\});
    form.text("{}Tabbed Panel demo"{});

    // construct tabbed panel
